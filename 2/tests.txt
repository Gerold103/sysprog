----# Test { python script create and run --------------------------------------

time.sleep(0.1)\n\
f = open('test.txt', 'a')\n\
f.write('Text\\\\n')\n\
f.close()\n" > test.py

python3 test.py | exit 0

cat test.txt
----# Output
Text
----# }

----# Test { same non-idempotent commands --------------------------------------
rm test.txt
printf "with open('test.txt', 'a') as myfile:\n\
    myfile.write('a\\\\n')\n\
" > test.py

python3 test.py | python3 test.py | python3 test.py | python3 test.py

cat test.txt
----# Output
a
a
a
a
----# }
